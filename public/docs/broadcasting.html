# Broadcasting Guide > Real-time events in TAVP Stack. ## Configuration ```php // config/broadcasting.php return [ 'default' => env('BROADCAST_DRIVER', 'log'), 'connections' => [ 'pusher' => [ 'driver' => 'pusher', 'key' => env('PUSHER_APP_KEY'), 'secret' => env('PUSHER_APP_SECRET'), 'app_id' => env('PUSHER_APP_ID'), 'options' => [ 'cluster' => env('PUSHER_APP_CLUSTER'), 'useTLS' => true, ], ], 'soketi' => [ 'driver' => 'soketi', 'key' => env('SOKETI_APP_KEY'), 'secret' => env('SOKETI_APP_SECRET'), 'app_id' => env('SOKETI_APP_ID'), 'host' => env('SOKETI_HOST', '127.0.0.1'), 'port' => env('SOKETI_PORT', 6001), ], 'redis' => [ 'driver' => 'redis', 'connection' => 'default', ], 'log' => [ 'driver' => 'log', ], ], ]; ``` ## Defining Events ```php namespace App\Events; use Tavp\Events\Broadcastable; use Tavp\Events\InteractsWithSockets; class NewMessage extends Broadcastable { public $message; public function __construct($message) { $this->message = $message; } public function broadcastOn() { return new Channel('messages'); } } ``` ## Broadcasting Events ```php use App\Events\NewMessage; // Broadcast event event(new NewMessage($message)); // Broadcast to specific channel broadcast(new NewMessage($message))->toOthers(); ``` ## Listening for Events (JavaScript) ```javascript // Using Pusher.js or Soketi import Echo from 'laravel-echo'; import Pusher from 'pusher-js'; window.Echo = new Echo({ broadcaster: 'pusher', key: 'your-app-key', cluster: 'mt1', }); // Listen for events Echo.channel('messages') .listen('NewMessage', (e) => { console.log(e.message); }); ``` ## Private Channels ```php // Define private channel public function broadcastOn() { return new PrivateChannel('user.' . $this->user->id); } // Authorize channel $router->post('/api/broadcasting/auth', function () { return auth()->user(); }); ``` ## Presence Channels ```php // Define presence channel public function broadcastOn() { return new PresenceChannel('room.' . $this->room->id); } // Get presence data public function broadcastWith() { return [ 'user' => $this->user, 'message' => $this->message, ]; } ``` ## Drivers Comparison | Driver | Speed | Cost | Scalability | |--------|-------|------|-------------| | Pusher | Fast | Paid | High | | Soketi | Fast | Free | High | | Redis | Fast | Free | High | | Log | Slow | Free | Low |
