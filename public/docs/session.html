# Session Guide > Session management in TAVP Stack. ## Configuration ```php // config/session.php return [ 'driver' => env('SESSION_DRIVER', 'file'), 'lifetime' => 120, 'expire_on_close' => false, 'encrypt' => false, 'files' => storage_path('framework/sessions'), 'connection' => env('SESSION_CONNECTION'), 'table' => 'sessions', 'store' => env('SESSION_STORE'), 'lottery' => [2, 100], 'cookie' => env('SESSION_COOKIE', 'tavp_session'), 'path' => '/', 'domain' => env('SESSION_DOMAIN'), 'secure' => env('SESSION_SECURE_COOKIE'), 'http_only' => true, 'same_site' => 'lax', ]; ``` ## Basic Usage ```php use Tavp\Support\Facades\Session; // Store value Session::put('key', 'value'); // Get value $value = Session::get('key'); // Get with default $value = Session::get('key', 'default'); // Check if exists if (Session::has('key')) { // Value exists } // Forget value Session::forget('key'); // Flush all Session::flush(); // Pull value (get and remove) $value = Session::pull('key'); ``` ## Flash Data ```php // Flash data for next request Session::flash('success', 'Operation successful!'); // Reflash all flash data Session::reflash(); ``` ## Session in Controllers ```php namespace App\Controllers; use Tavp\Http\Controller; class UserController extends Controller { public function store() { // ... $this->session->set('success', 'User created!'); return redirect('/users'); } } ``` ## Session in Views ```volt {% if session('success') %}
{{ session('success') }}
{% endif %} ```
