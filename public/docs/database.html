Database Guide

Configuration

Configure your database in config/database.php:

return [
    'default' => env('DB_CONNECTION', 'mysql'),
    
    'connections' => [
        'mysql' => [
            'driver' => 'mysql',
            'host' => env('DB_HOST', '127.0.0.1'),
            'port' => env('DB_PORT', '3306'),
            'database' => env('DB_DATABASE', 'tavp'),
            'username' => env('DB_USERNAME', 'root'),
            'password' => env('DB_PASSWORD', ''),
            'charset' => 'utf8mb4',
            'collation' => 'utf8mb4_unicode_ci',
        ],
        
        'sqlite' => [
            'driver' => 'sqlite',
            'database' => env('DB_DATABASE', database_path('database.sqlite')),
        ],
        
        'pgsql' => [
            'driver' => 'pgsql',
            'host' => env('DB_HOST', '127.0.0.1'),
            'database' => env('DB_DATABASE', 'tavp'),
            'username' => env('DB_USERNAME', 'postgres'),
            'password' => env('DB_PASSWORD', ''),
        ],
    ],
];

Running Migrations

# Run all pending migrations
tavp migrate

# Rollback last migration
tavp migrate:rollback

# Rollback all migrations
tavp migrate:reset

# Fresh start (drop all tables + re-migrate)
tavp migrate:fresh

# Fresh start with seeding
tavp migrate:fresh --seed

Raw Queries

use Tavp\Support\Facades\DB;

// Select
$users = DB::select('SELECT * FROM users WHERE active = ?', [1]);

// Insert
DB::insert('INSERT INTO users (name, email) VALUES (?, ?)', ['John', 'john@example.com']);

// Update
DB::update('UPDATE users SET active = ? WHERE id = ?', [0, 1]);

// Delete
DB::delete('DELETE FROM users WHERE id = ?', [1]);

// Transactions
DB::transaction(function () {
    DB::update('UPDATE accounts SET balance = balance - ? WHERE id = ?', [100, 1]);
    DB::update('UPDATE accounts SET balance = balance + ? WHERE id = ?', [100, 2]);
});
